    public boolean isValidText( String s )

        return  (s != null && !s.isBlank()) ;
    }
    public boolean isValidNumber( String s )
    {
        return isValidInteger(s) ||
                isValidLong(s)||
                isValidDouble(s);

            {
                return_value = false;

        } else return_value = false ;
        return return_value ;
    }
